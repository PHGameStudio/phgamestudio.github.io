# 特乐葫芦

<img id="xhl-img" src="/easter-eggs/xaba-hs-lou.png" alt="" style="width:85%" />

<script>
(function() {
var imgs = [
{ src: '/easter-eggs/xaba-hs-lou.png', weight: 1 },
{ src: '/easter-eggs/xaba-hs-lou-2.jpg', weight: 1 },
{ src: '/easter-eggs/xaba-hs-lou-3.jpg', weight: 1 },
];
var totalWeight = 0;
for (var i = 0; i < imgs.length; i++) {
totalWeight += imgs[i].weight;
}
var pick = imgs[0].src;
var random = Math.random() * totalWeight;
for (var i = 0; i < imgs.length; i++) {
random -= imgs[i].weight;
if (random < 0) {
pick = imgs[i].src;
break;
}
}
var el = document.getElementById('xhl-img');
if (el) el.src = pick;
console.log("正在随机把一张特乐葫芦的图片选择出来")
})();
</script>
